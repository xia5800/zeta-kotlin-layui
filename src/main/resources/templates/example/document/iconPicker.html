layui.use(['iconPicker'], function() {
    var iconPicker = layui.iconPicker;

    iconPicker.render({
        elem: '#iconPicker',
        type: 'fontClass',
        search: true,
        page: true,
        limit: 16,
        click: function(data) {
           console.log(data);
        },
        success: function(d) {
            console.log(d);
        }
    });

});
